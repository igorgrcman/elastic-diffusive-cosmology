%% ===== INLINE STOPLIGHT LEGEND =====
\newcommand{\stoplightLegendInline}{%
  \textcolor{OliveGreen}{\textbf{GREEN}} = derived/verified \quad
  \textcolor{YellowOrange}{\textbf{YELLOW}} = mechanism identified \quad
  \textcolor{BrickRed}{\textbf{RED}} = open/falsified%
}

%% ===== FULL STOPLIGHT LEGEND (table format) =====
\newcommand{\stoplightLegendTable}{%
  \begin{center}
  \small
  \begin{tabular}{@{}lp{10cm}@{}}
  \toprule
  \textbf{Status} & \textbf{Meaning} \\
  \midrule
  \textcolor{OliveGreen}{\textbf{GREEN}} &
    Claim is derived from EDC postulates with explicit calculation.
    Evidence includes equations with [\texttt{Der}] or [\texttt{Dc}] tags.
    Numerical agreement $<5\%$ with experiment. \\[0.5em]
  \textcolor{YellowOrange}{\textbf{YELLOW}} &
    Mechanism identified but quantitative closure incomplete.
    May involve phenomenological ansatz [\texttt{P}] or calibration [\texttt{Cal}].
    Order-of-magnitude agreement only. \\[0.5em]
  \textcolor{BrickRed}{\textbf{RED}} &
    Open problem: derivation not achieved.
    May be falsified (baseline attempt fails) or simply not attempted.
    Requires future work. \\[0.5em]
  \sout{\textbf{FALSIFIED}} &
    Specific approach was tested and definitively shown to fail.
    Documents what does NOT work (valuable negative result). \\
  \bottomrule
  \end{tabular}
  \end{center}
}

%% ===== COMPACT LEGEND (for margin or footnote) =====
\newcommand{\stoplightLegendCompact}{%
  \footnotesize
  \textcolor{OliveGreen}{$\blacksquare$}~derived\quad
  \textcolor{YellowOrange}{$\blacksquare$}~partial\quad
  \textcolor{BrickRed}{$\blacksquare$}~open%
}

%% ===== EPISTEMIC TAG LEGEND =====
\newcommand{\epistemicTagLegend}{%
  \begin{center}
  \small
  \begin{tabular}{@{}ll@{}}
  \toprule
  \textbf{Tag} & \textbf{Meaning} \\
  \midrule
  \textsf{[Der]} & Derived: explicit derivation from EDC postulates \\
  \textsf{[Dc]} & Derived-computed: numerical evaluation of derived expression \\
  \textsf{[P]} & Proposed: phenomenological ansatz or conjecture \\
  \textsf{[I]} & Identified: pattern matching (not unique derivation) \\
  \textsf{[Cal]} & Calibrated: parameter fitted to data \\
  \textsf{[BL]} & Baseline: external reference (PDG, CODATA) \\
  \bottomrule
  \end{tabular}
  \end{center}
}
