%% edc_intermezzo_boxes.tex — Styled boxes for intermezzo conclusions
%% Created: 2026-01-22
%% Purpose: Consistent intermezzo formatting across EDC documents
%% Requires: tcolorbox package

%% ===== INTERMEZZO BOX STYLES =====

%% Standard Intermezzo box (neutral)
\newtcolorbox{intermezzoBox}[1][]{%
  enhanced,
  colback=gray!5,
  colframe=gray!50,
  fonttitle=\bfseries,
  title=Intermezzo,
  #1
}

%% Success Intermezzo (GREEN conclusion)
\newtcolorbox{intermezzoGreen}[1][]{%
  enhanced,
  colback=OliveGreen!5,
  colframe=OliveGreen!70,
  fonttitle=\bfseries,
  title=Intermezzo — Conclusion Established,
  #1
}

%% Partial Intermezzo (YELLOW — mechanism identified)
\newtcolorbox{intermezzoYellow}[1][]{%
  enhanced,
  colback=YellowOrange!5,
  colframe=YellowOrange!70,
  fonttitle=\bfseries,
  title=Intermezzo — Partial Progress,
  #1
}

%% Falsification Intermezzo (RED — attempt failed)
\newtcolorbox{intermezzoRed}[1][]{%
  enhanced,
  colback=BrickRed!5,
  colframe=BrickRed!70,
  fonttitle=\bfseries,
  title=Intermezzo — Falsified / Open,
  #1
}

%% ===== CHAIN BOX (derivation chain summary) =====
\newtcolorbox{chainBox}[1][]{%
  enhanced,
  colback=blue!3,
  colframe=blue!40,
  fonttitle=\bfseries,
  title=Chain Box — Derivation Summary,
  #1
}

%% ===== CLAIM BOX (single claim with status) =====
%% Usage: \begin{claimBox}[title=Claim X.Y] ... \end{claimBox}
\newtcolorbox{claimBox}[1][]{%
  enhanced,
  colback=white,
  colframe=black!40,
  fonttitle=\bfseries\small,
  boxrule=0.5pt,
  left=3pt,
  right=3pt,
  top=2pt,
  bottom=2pt,
  #1
}
