%% edc_meta_macros.tex — Macros for EDC meta-documentation
%% Created: 2026-01-22
%% Purpose: Consistent formatting for claim ledgers, decision logs, timelines

%% ===== CLAIM STATUS COLORS =====
\newcommand{\claimGreen}[1]{\textcolor{OliveGreen}{\textbf{GREEN}}: #1}
\newcommand{\claimYellow}[1]{\textcolor{YellowOrange}{\textbf{YELLOW}}: #1}
\newcommand{\claimRed}[1]{\textcolor{BrickRed}{\textbf{RED}}: #1}
\newcommand{\claimFalsified}[1]{\textcolor{gray}{\sout{\textbf{FALSIFIED}}}: #1}

%% ===== EPISTEMIC TAG FORMATTING =====
%% Use \providecommand to avoid conflicts with main document definitions
\providecommand{\tagDer}{\textsf{[Der]}}
\providecommand{\tagDc}{\textsf{[Dc]}}
\providecommand{\tagP}{\textsf{[P]}}
\providecommand{\tagBL}{\textsf{[BL]}}
\providecommand{\tagCal}{\textsf{[Cal]}}
\providecommand{\tagI}{\textsf{[I]}}

%% ===== CLAIM ENTRY FORMATTING =====
%% Usage: \claimEntry{ID}{Status}{Chapter}{Description}{Evidence}{Tag}
\newcommand{\claimEntry}[6]{%
  \noindent\textbf{#1} \hfill \textit{#2}\\
  \textbf{Chapter:} #3\\
  \textbf{Claim:} #4\\
  \textbf{Evidence:} #5 \hfill #6\\[0.5em]
  \hrule\vspace{0.5em}%
}

%% ===== DECISION LOG ENTRY =====
%% Usage: \decisionEntry{ID}{Date}{Decision}{Rationale}{Alternatives}{Commit}
\newcommand{\decisionEntry}[6]{%
  \noindent\textbf{#1} \hfill \textit{#2}\\
  \textbf{Decision:} #3\\
  \textbf{Rationale:} #4\\
  \textbf{Alternatives considered:} #5\\
  \textbf{Commit:} \texttt{#6}\\[0.5em]
  \hrule\vspace{0.5em}%
}

%% ===== TIMELINE ENTRY =====
%% Usage: \timelineEntry{Date}{Event}{Files}{Commit}
\newcommand{\timelineEntry}[4]{%
  \noindent\textbf{#1:} #2\\
  \textit{Files:} \texttt{#3} \hfill \texttt{#4}\\[0.3em]
}

%% ===== EVIDENCE LINK =====
%% Usage: \evidenceRef{file}{line}
\newcommand{\evidenceRef}[2]{\texttt{#1:#2}}

%% ===== INTERMEZZO REFERENCE =====
%% Usage: \intermezzoRef{chapter}{label}
\newcommand{\intermezzoRef}[2]{Intermezzo~#1 (§\ref{#2})}

%% ===== VERSION MARKER =====
\newcommand{\metaVersion}[1]{\textit{Meta-documentation v#1}}
