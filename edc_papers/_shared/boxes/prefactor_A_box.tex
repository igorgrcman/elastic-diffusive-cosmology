\textbf{Derived formula:}
\begin{equation*}
    \boxed{A = \frac{\pi \omega_0}{\omega_B \sqrt{L_0/\delta}} = 1.03 \times \frac{\omega_0}{\omega_B}}
\end{equation*}

\textbf{Parameters:}
\begin{itemize}
    \item $\omega_0 = \sqrt{\sigma/m_p} = 19.1$ MeV --- well frequency \tagDc{}
    \item $\omega_B = \sqrt{|V''(q_B)|/M} \approx 23$ MeV --- barrier frequency \tagDc{}
    \item $\omega_0/\omega_B \approx 0.82$ --- barrier is $\sim 22\%$ steeper than well
\end{itemize}

\textbf{Result:} $A \approx 0.84$ \tagDer{} (within 1D effective theory)

\textbf{Physical insight:} The prefactor $A < 1$ arises because the barrier curvature exceeds the well curvature ($\omega_B > \omega_0$).
