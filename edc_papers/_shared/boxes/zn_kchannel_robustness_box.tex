\begin{tcolorbox}[colback=green!5!white, colframe=green!50!black,
                  title=\textbf{Z$_N$ k-channel: Definition, Applicability \& Robustness}]
\label{box:zn-kchannel-robustness}

\textbf{Definition} (discrete/continuum averaging ratio):
\begin{equation}
k(N) := \frac{\langle O \rangle_{\text{disc}}}{\langle O \rangle_{\text{cont}}}
= 1 + \frac{1}{N} \quad \text{[Der]}
\end{equation}
Valid for observables with Z$_N$-symmetric anisotropy $O(\theta) = c + a\cos(N\theta)$
under equal-corner-share normalization $a/c = 1/N$.
For Z$_6$: $k(6) = 7/6$.

\vspace{0.3em}
\textbf{Applicability Rule} (CRITICAL):
\begin{itemize}[nosep,leftmargin=1.5em]
\item[\checkmark] \textbf{Use} for averaging observables approximated by discrete sampling
      (e.g., $N_{\text{cell}}$ bare$\to$effective, pion-like $\varepsilon$-dressing)
\item[$\times$] \textbf{Do NOT use} for cardinality/subgroup ratios
      ($\sin^2\theta_W = |Z_2|/|Z_6|$, $N_g = |Z_3|$, CP phases, Koide $Q$)
\end{itemize}

\vspace{0.3em}
\textbf{Robustness Results} [Der]:
\begin{enumerate}[nosep,leftmargin=1.5em]
\item \textit{Non-quadratic $W(u)$:} Mode index $m=N$ unchanged near equilibrium;
      nonlinearities add harmonics $2N, 3N, \ldots$ but do not shift leading mode.
\item \textit{Strong pinning ($\rho \gg N^2$):} Z$_N$ irrep structure and periodicity
      stable at any $\rho$. Mode shape (Fourier overlap) degrades but selection is protected.
\item \textit{One-defect ($\varepsilon \ne 0$):} Contamination $\sim O(\varepsilon^2)$;
      small defects ($|\varepsilon| < 10\%$) cause $<1\%$ overlap loss in weak/moderate pinning.
\end{enumerate}
