\section{Q Factorization From Action}
\label{app:qfactor}

\subsection{Part A: Y-Junction Degrees of Freedom}

\textbf{Definition \Dd:} Each flux tube $i \in \{1,2,3\}$ has:
\begin{itemize}
\item Position embedding: $X_i^\mu(s)$ --- worldline in 4D spacetime
\item Internal orientation: $\theta_i \in S^3 \cong \mathrm{SU}(2)$
\end{itemize}

\textbf{Junction Constraints \Dd:}
\begin{enumerate}
\item Position matching: $X_1(0) = X_2(0) = X_3(0) = X_J$
\item Tension balance: $\sum T_i \hat{n}_i = 0$
\item \textbf{No orientation constraint} (to be proven)
\end{enumerate}

\subsection{Part B: Route 1 --- Constraint Counting}

\textbf{Theorem (Orientation Independence) \Dc:}
Junction constraints impose NO relations on $\{\theta_1, \theta_2, \theta_3\}$.

\textbf{Proof:}
\begin{enumerate}
\item Position matching constrains WHERE tubes meet, not internal orientation
\item Tension balance constrains TANGENT directions $dX/ds$, not $S^3$ coordinates
\item Orientation $\theta_i$ describes extension into extra dimension --- geometrically independent
\item No physical mechanism couples orientations at junction
\end{enumerate}

\subsection{Part C: Route 2 --- Path Integral}

\textbf{Proposition \Pp:} The total action decomposes as:
\[
S_{\mathrm{total}} = S_{\mathrm{position}}[X_1, X_2, X_3] + \sum_{i=1}^3 S_{\mathrm{orient}}[\theta_i]
\]

\textbf{Theorem (Measure Factorization) \Dc:}
If $S_{\mathrm{total}}$ has no $\theta_i$-$\theta_j$ cross-terms:
\[
d\mu_Q = d\mu_1 \otimes d\mu_2 \otimes d\mu_3
\]
