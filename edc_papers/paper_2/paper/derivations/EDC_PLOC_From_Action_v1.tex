\section{P-loc From Action}
\label{app:ploc}

\subsection{Part A: The Localization Statement}

\textbf{Definition (5D Density) \Dd:}
\[
\rho_5(x,\xi) = |\Psi(x,\xi)|^2
\]

\textbf{P-loc \Pp{} $\to$ \Dc:} The electron 5D density factorizes as:
\[
\rho_5(x,\xi) = \rho_4(x) \cdot f(\xi) \quad \text{with } f(\xi) \to \delta(\xi)
\]

\subsection{Part B: Route 2 --- Thin-Brane Limit (Preferred)}

\textbf{Step T1 \Dd:} Membrane at $\xi = 0$ has physical thickness characterized by tension $\sigma$.

\textbf{Step T2 \Dc:} Dimensional analysis gives membrane thickness:
\[
w \sim \sqrt{\frac{\hbar c}{\sigma}}
\]

\textbf{Step T3 \Dc:} Particles confined to membrane inherit its profile. Effective $\kappa$:
\[
\kappa_{\mathrm{eff}} \sim \frac{\sigma^2}{\hbar c}
\]

\textbf{Step T4 \Pp:} P-$\sigma$ states that $\sigma$ is large ($\sigma\Delta A \gg \hbar$).

\textbf{Step T5 \Dc:} Large $\sigma$ implies:
\begin{itemize}
\item Large $\kappa_{\mathrm{eff}}$
\item Small width $w \sim \sigma^{-1/2}$
\item Therefore: $f(\xi) \to \delta(\xi)$
\end{itemize}
